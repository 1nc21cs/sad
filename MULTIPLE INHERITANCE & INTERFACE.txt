import java.util.*;

interface Itemcount {
    public void Item();
}

class product {
    String producttype = "Hard ware";
    int productnumber = 123;
}

class Softwareproducts extends product implements Itemcount {
    int java = 100;
    int DS = 60;
    int stotal;

    public void scalculate() {
        stotal = java + DS;
    }

    public void Item() {
        System.out.println("the number of Softwareproduct is:" + stotal);
    }
}

class Hardwareproduct extends Softwareproducts implements Itemcount {
    int printer = 20;
    int speaker = 25;
    int htotal;

    public void hcalculate() {
        htotal = printer + speaker;
    }

    public void Items() {
        System.out.println("\nthe number of Hardwareproduct is:" + htotal);
        System.out.println("\nthe producttype is:" + producttype);
        System.out.println("\nthe productnumber is:" + productnumber);
    }

    public static void main(String args[]) {
        Softwareproducts s = new Softwareproducts();
        Hardwareproduct h = new Hardwareproduct();
        s.scalculate();
        h.hcalculate();
        s.Item();
        h.Items();
    }
}

/* output of this program is:

the number of Softwareproduct is:160

the number of Hardwareproduct is:45

the producttype is:Hard ware

the productnumber is:123
*/
