import java.applet.*;
import java.awt.*;
public class First extends Applet
{
 int length, breadth;
 public void init( ) 
 {
System.out.println("Welcome to Init method "); 
setBackground(Color.green);
setForeground(Color.black);
length=10;
breadth=30;
 }
public void start( ) 
 {
 System.out.println("Welcome to Start method "); 
 }
public void paint(Graphics g) 
 {
 System.out.println("Welcome to Paint method ");
 int area=length*breadth;
 g.drawString("Area of Rectangle is "+area ,100,100);
 }
public void stop( )
 {
 System.out.println("Welcome to Stop method ");
 }
public void destroy( )
 {
 System.out.println("Welcome to destroy method "); 
 }
}
