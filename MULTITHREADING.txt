import java.util.*;
public class Multithreading {
public static void main(String[ ] args)
{
first t1=new first( ); 
t1.start( ) ;
}
}
class first extends Thread
{
public void run( )
{
int num=0;
Random r=new Random(); 
try
{
for(int i=1;i<=3;i++)
{
num=r.nextInt(10);
System.out.println("first thread generated num is ="+num);
Thread t2 = new Thread(new second(num)); 
t2.start();

Thread t3 = new Thread(new third(num)); 
t3.start();
Thread.sleep(2000);
}
}
catch(Exception e)
{
System.out.println(e.getMessage());
}
}
}
class second implements Runnable
{
int x;
public second(int x)
{
this.x=x;
}

public void run()
{
System.out.println("\nSecond thread: Square of 2 num is " +(x*x));
}
}

class third implements Runnable
{
public int x; 
public third(int x)
{
this.x=x;
}
public void run()
{
System.out.println("\nThird thread: Cube of num is " +(x*x*x));
}
}
